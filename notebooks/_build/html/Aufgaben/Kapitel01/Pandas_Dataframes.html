import pandas as pd

df = pd.DataFrame(
    {
        "id": [1, 2, 3, 4],
        "Name": ["John", "Alice", "Mike", "Anne"],
        "Age": [26, 20, 21, 25],
    }
)
df

0 1 John 26
1 2 Alice 20
2 3 Mike 21
3 4 Anne 25
